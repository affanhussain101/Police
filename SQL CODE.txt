CREATE TABLE STATION(
STATION_ID INT,
NAME VARCHAR(30),
ADDRESS VARCHAR(60),
PRIMARY KEY(BRANCH_ID)
);

CREATE TABLE POLICE(
POLICE_ID INT,
NAME VARCHAR(30),
DESIGNATION VARCHAR(30),
CONTACT_NO BIGINT,
STATION_ID INT,
PRIMARY KEY (POLICE_ID),
FOREIGN KEY FK1 (STATION_ID) REFERENCES STATION(STATION_ID)
);

CREATE TABLE CRIME(
CRIME_ID INT,
TYPE VARCHAR(10),
PRIMARY KEY(CRIME_ID));

CREATE TABLE FIR(
FIR_NO VARCHAR(10),
DATE DATE,
INFORMANT_NAME VARCHAR(30),
CRIME_PLACE VARCHAR(60),
CIRME_ID INT,
PRIMARY KEY(FIR_NO),
FOREIGN KEY FK2 (CRIME_ID) REFERENCES CRIME(CRIME_ID)
);

CREATE TABLE CRIMINAL(
NAME VARCHAR(30),
CRIMINAL_ID INT,
ADDRESS VARCHAR(30),
CRIME_ID INT,
FIR_NO VARCHAR(10),
POLICE_ID INT,
PRIMARY KEY (CRIMINAL_ID),
FOREIGN KEY FK3 (CRIME_ID) REFERENCES CRIME(CRIME_ID),
FOREIGN KEY FK4 (FIR_NO) REFERENCES FIR(FIR_NO),
FOREIGN KEY FK5 (POLICE_ID) REFERENCES POLICE(POLCE_ID)
);


CREATE TABLE DOES(
CRIME_ID INT,
CRIMINAL_ID INT,
FOREIGN KEY FK6 (CRIME_ID) REFERENCES CRIME(CRIME_ID),
FOREIGN KEY FK7 (CRIMINAL_ID) REFERENCES CRIMINAL(CRIMINAL_ID)
);
